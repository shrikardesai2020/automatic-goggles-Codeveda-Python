import random

num = random.randint(1,100)
attempts = 7

print("Guess number between 1 and 100")

while attempts>0:
    guess = int(input("Enter guess: "))

    if guess==num:
        print("Correct!")
        break
    elif guess>num:
        print("Too high")
    else:
        print("Too low")

    attempts -= 1

if attempts==0:
    print("Game Over. Number was", num)
