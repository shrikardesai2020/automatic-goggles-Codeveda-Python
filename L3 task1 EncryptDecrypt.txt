def encrypt(text,shift):
    res=""
    for c in text:
        if c.isalpha():
            res+=chr((ord(c)+shift-97)%26+97)
        else:
            res+=c
    return res

def decrypt(text,shift):
    return encrypt(text,-shift)

msg=input("Message:")
s=int(input("Shift:"))

enc=encrypt(msg,s)
print("Encrypted:",enc)
print("Decrypted:",decrypt(enc,s))
